
  SELECT
   start_station_name AS subscribber_casual,
   round(COUNT(trip_id)*100/983648,2) AS ct_st
 FROM
   `bigquery-public-data.san_francisco.bikeshare_trips`
 WHERE
   duration_sec >= 1800
   AND subscriber_type = "Subscriber"
 GROUP BY
   start_station_name
 ORDER BY
   ct_st DESC
 LIMIT
   5
